	integer thirdNumber;
	integer minimum = 0;

	$
		minimum = firstNumber;

	$

		$
			minimum = secondNumber;
		$
		else
		$
			minimum = thirdNumber;
		$
	$
	
	writeToKeyboard << "Minimum: " << minimum;
	
	return 0;
